Usage


import pizzle from 'pizzle';
var animate = new pizzle.create()
animate.init()

For Customization

Default Setting

import pizzle from 'pizzle';
var animate = new pizzle.create({
  target: "app", // target for the pizzle-start attribute
  duration: 1000, // for the animation duration fallback
  directives: {} // for creating custom directives. Well not that useful
})
animate.init()

import pizzle from 'pizzle';
var animate = new pizzle.create({
  target: "boom",
  duration: 4000,
  directives:{
     html: function(el,{bind,name}){
      // el is the target element
      // bind is the value of the directives
      // name is the name of the directives
      el.innerHTML = bind;
    }
  }
})
animate.init()

<!-- Use the directives in the markup like this

  Note we have override the default target value so we have to change it in the pizzle-start attribute
-->
<div pizzle-start="boom">
  <h1 pizzle:html="Hello World"></h1>
  <!-- Using default values -->
  <h1 pizzle:html="Hello World" pizzle-bounce="infinite"></h1>
  <!-- You want the animation to run once -->
  <h1 pizzle:html="Hello World" pizzle-bounce></h1>
</div>

Create Your own pizzle animation

In this format

@keyframes rotate{
  from{
    transform: rotate(0deg);
  }
  to{
    transform: rotate(360deg)
  }
}
.rotate{
  animation-name: rotate;
  animation-timing-function: linear;
}
.rotate-1000{
  animation-duration: 1s;
}
.rotate-2000{
  animation-duration: 2s;
}
.rotate-3000{
  animation-duration: 3s;
}
.rotate-4000{
  animation-duration: 4s;
}
.rotate-5000{
  animation-duration: 5s;
}
.rotate-infinite{
  animation-iteration-count: infinite;
}
.rotate-reverse{
  animation-direction: reverse;
}
